package portfolio

import "github.com/penny-vault/pvbt/asset"

// ExcursionRecord tracks the running price extremes for an open position.
// EntryPrice is the fill price of the first buy that opened the position.
// HighPrice tracks the running maximum of daily High prices observed while
// the position is open. LowPrice tracks the running minimum of daily Low
// prices. These are used to compute MFE and MAE when the position closes.
type ExcursionRecord struct {
    EntryPrice float64
    HighPrice  float64
    LowPrice   float64
}

In `portfolio/excursion.go`:

```go
// UpdateExcursions reads the daily High and Low prices from the DataFrame
// and updates the running extremes for each open position. If High or Low
// is NaN for an asset, that asset is skipped for that day.
func (a *Account) UpdateExcursions(df *data.DataFrame) {
    for ast := range a.excursions {
        high := df.Value(ast, data.MetricHigh)
        low := df.Value(ast, data.MetricLow)

        rec := a.excursions[ast]

        if !math.IsNaN(high) && high > rec.HighPrice {
            rec.HighPrice = high
        }

        if !math.IsNaN(low) && low < rec.LowPrice {
            rec.LowPrice = low
        }

        a.excursions[ast] = rec
    }
}
```

Add import for `"math"` and `"github.com/penny-vault/pvbt/data"` to `excursion.go`.

    EntryPrice float64
    ExitPrice  float64
    Qty        float64
    PnL        float64
    HoldDays   float64
    MFE        float64
    MAE        float64
}

- [ ] **Step 5: Produce TradeDetail on sell in Record()**

In `Record()`, inside the `SellTransaction` case, before the existing FIFO lot matching, compute TradeDetails from the excursion record. The FIFO matching for `TradeDetail` mirrors the existing `taxLots` FIFO but produces `TradeDetail` entries.

Add this logic in the `SellTransaction` case of `Record()`, after the line `a.holdings[txn.Asset] -= txn.Qty` but before the existing tax lot FIFO loop:

```go
// Produce TradeDetail entries from excursion data.
if exc, hasExc := a.excursions[txn.Asset]; hasExc {
    mfe := (exc.HighPrice - exc.EntryPrice) / exc.EntryPrice
    mae := (exc.LowPrice - exc.EntryPrice) / exc.EntryPrice

    // Match against tax lots FIFO to get entry dates and per-lot qty.
    remaining := txn.Qty
    lots := a.taxLots[txn.Asset]
    for idx := 0; idx < len(lots) && remaining > 0; idx++ {
        matched := lots[idx].Qty
        if matched > remaining {
            matched = remaining
        }

        a.tradeDetails = append(a.tradeDetails, TradeDetail{
            Asset:      txn.Asset,
            EntryDate:  lots[idx].Date,
            ExitDate:   txn.Date,
            EntryPrice: lots[idx].Price,
            ExitPrice:  txn.Price,
            Qty:        matched,
            PnL:        (txn.Price - lots[idx].Price) * matched,
            HoldDays:   txn.Date.Sub(lots[idx].Date).Hours() / 24.0,
            MFE:        mfe,
            MAE:        mae,
        })

        remaining -= matched
    }
}
```

Note: this reads `taxLots` before the existing FIFO loop consumes them, so it must come before the existing lot matching code.

package portfolio

import "math"

type tradeCaptureRatio struct{}

func (tradeCaptureRatio) Name() string { return "TradeCaptureRatio" }

func (tradeCaptureRatio) Description() string {
    return "Ratio of mean realized return to mean MFE. A value of 1.0 means the strategy " +
        "captures all available favorable movement on average. Lower values indicate " +
        "premature exits that leave profit on the table."
}

func (tradeCaptureRatio) Compute(a *Account, _ *Period) (float64, error) {
    details := a.TradeDetails()
    if len(details) == 0 {
        return math.NaN(), nil
    }

    var sumReturn, sumMFE float64
    for _, td := range details {
        sumReturn += (td.ExitPrice - td.EntryPrice) / td.EntryPrice
        sumMFE += td.MFE
    }

    avgReturn := sumReturn / float64(len(details))
    avgMFE := sumMFE / float64(len(details))

    if avgMFE == 0 {
        return math.NaN(), nil
    }

    return avgReturn / avgMFE, nil
}

func (tradeCaptureRatio) ComputeSeries(a *Account, window *Period) ([]float64, error) {
    return nil, nil
